# Vim config for windows

## INSTALL DETAILS WINDOWS

cd ~

git clone https://github.com/Snaptags/vimconfig vimfiles

cd vimfiles

git submodule init

git submodule update


## INSTALL DETAILS LINUX

cd ~

git clone https://github.com/Snaptags/vimconfig .vim

cd .vim

git submodule init

git submodule update
